67// Blocks.
68//中文注释：
69// 块。
70// Allocate a zeroed disk block.
71//中文注释：
72// 分配一个清零的磁盘块。
73static uint balloc(uint dev)//分配块函数，参数为设备号
74{
75 int b, bi, m;//循环变量和掩码
76 struct buf *bp;//缓冲区指针
77
78 //以下代码用于遍历超级块中的所有块，查找一个未被使用的块，并将其标记为已使用，然后返回该块的块号。
79//具体实现如下：
80// 遍历超级块中的所有块，超级块包含文件系统的元数据，包括块数、inode数等。
81// 每次处理BPB个块，BPB，即每个块位图块可以表示的块数，位图块用于跟踪哪些块是已分配的，哪些是空闲的。
82// 读取当前块对应的位图块到缓冲区
83// 遍历位图块中的每个位，检查对应的块是否为空闲
84// 如果找到一个空闲块，标记为已使用，写回日志，释放缓冲区，清零该块，并返回块号
85// 如果没有找到空闲块，释放缓冲区，继续查找下一个块
86
87 bp = 0;
88 for(b = 0; b < sb.size; b += BPB){
89 bp = bread(dev, BBLOCK(b, sb));
90 for(bi = 0; bi < BPB && b + bi < sb.size; bi++){
91 m = 1 << (bi % 8);
92 if((bp->data[bi/8] & m) == 0){ // Is block free?
93 bp->data[bi/8] |= m; // Mark block in use.
94 log_write(bp);
95 brelse(bp);
96 bzero(dev, b + bi);
97 return b + bi;
98 }
99 }
100 brelse(bp);
101 }
102 panic("balloc: out of blocks");
103}

105// Free a disk block.
106static void bfree(int dev, uint b)
107{
108 struct buf *bp;
109 int bi, m;
110
111 bp = bread(dev, BBLOCK(b, sb));
112 bi = b % BPB;
113 m = 1 << (bi % 8);

115 panic("freeing free block");
116 bp->data[bi/8] &= ~m;
117 log_write(bp);
118 brelse(bp);

121// Inodes.
122//
123// An inode describes a single unnamed file.
124// The inode disk structure holds metadata: the file's type,
125// its size, the number of links referring to it, and the
126// list of blocks holding the file's content.
127//
128// The inodes are laid out sequentially on disk at
129// sb.startinode. Each inode has a number, indicating its
130// position on the disk.
131//
132// The kernel keeps a cache of in-use inodes in memory
133// to provide a place for synchronizing access
134// to inodes used by multiple processes. The cached
135// inodes include book-keeping information that is
136// not stored on disk: ip->ref and ip->valid.
137//
138// An inode and its in-memory representation go through a
139// sequence of states before they can be used by the
140// rest of the file system code.
141//
142// * Allocation: an inode is allocated if its type (on disk)
143// is non-zero. ialloc() allocates, and iput() frees if
144// the reference and link counts have fallen to zero.
145//
146// * Referencing in cache: an entry in the inode cache
147// is free if ip->ref is zero. Otherwise ip->ref tracks
148// the number of in-memory pointers to the entry (open
149// files and current directories). iget() finds or
150// creates a cache entry and increments its ref; iput()
151// decrements ref.
152//
153// * Valid: the information (type, size, &c) in an inode
154// cache entry is only correct when ip->valid is 1.
155// ilock() reads the inode from
156// the disk and sets ip->valid, while iput() clears
157// ip->valid if ip->ref has fallen to zero.
158//
159// * Locked: file system code may only examine and modify
160// the information in an inode and its content if it
161// has first locked the inode.
162//
163// Thus a typical sequence is:
164// ip = iget(dev, inum)
165// ilock(ip)
166// ... examine and modify ip->xxx ...
167// iunlock(ip)
168// iput(ip)
169//
170// ilock() is separate from iget() so that system calls can
171// get a long-term reference to an inode (as for an open file)
172// and only lock it for short periods (e.g., in read()).
173// The separation also helps avoid deadlock and races during
174// pathname lookup. iget() increments ip->ref so that the inode
175// stays cached and pointers to it remain valid.
176//
177// Many internal file system functions expect the caller to
178// have locked the inodes involved; this lets callers create
179// multi-step atomic operations.
180//
181// The icache.lock spin-lock protects the allocation of icache
182// entries. Since ip->ref indicates whether an entry is free,
183// and ip->dev and ip->inum indicate which i-node an entry
184// holds, one must hold icache.lock while using any of those fields.
185//
186// An ip->lock sleep-lock protects all ip-> fields other than ref,
187// dev, and inum. One must hold ip->lock in order to
188// read or write that inode's ip->valid, ip->size, ip->type, &c.
189
190struct {
191 struct spinlock lock;
192 struct inode inode[NINODE];
193} icache;
194
195void iinit(int dev)
196{
197 int i = 0;
198
199 initlock(&icache.lock, "icache");
200 for(i = 0; i < NINODE; i++) {
201 initsleeplock(&icache.inode[i].lock, "inode");
202 }
203
204 readsb(dev, &sb);
205 cprintf("sb: size %d nblocks %d ninodes %d nlog %d logstart %d\
206 inodestart %d bmap start %d\n", sb.size, sb.nblocks,
207 sb.ninodes, sb.nlog, sb.logstart, sb.inodestart,
208 sb.bmapstart);
209}

214// Allocate an inode on device dev.
215// Mark it as allocated by giving it type type.
216// Returns an unlocked but allocated and referenced inode.
217struct inode* ialloc(uint dev, short type)
218{
219 int inum;
220 struct buf *bp;
221 struct dinode *dip;
222
223 for(inum = 1; inum < sb.ninodes; inum++){
224 bp = bread(dev, IBLOCK(inum, sb));
225 dip = (struct dinode*)bp->data + inum%IPB;
226 if(dip->type == 0){ // a free inode
227 memset(dip, 0, sizeof(*dip));
228 dip->type = type;
229 log_write(bp); // mark it allocated on the disk
230 brelse(bp);
231 return iget(dev, inum);
232 }
233 brelse(bp);
234 }
235 panic("ialloc: no inodes");
236}

259// Find the inode with number inum on device dev
260// and return the in-memory copy. Does not lock
261// the inode and does not read it from disk.
262static struct inode* iget(uint dev, uint inum)
263{
264 struct inode *ip, *empty;
265
266 acquire(&icache.lock);
267
268 // Is the inode already cached?
269 empty = 0;
270 for(ip = &icache.inode[0]; ip < &icache.inode[NINODE]; ip++){
271 if(ip->ref > 0 && ip->dev == dev && ip->inum == inum){
272 ip->ref++;
273 release(&icache.lock);
274 return ip;
275 }
276 if(empty == 0 && ip->ref == 0) // Remember empty slot.
277 empty = ip;
278 }
279
280 // Recycle an inode cache entry.
281 if(empty == 0)
282 panic("iget: no inodes");
283
284 ip = empty;
285 ip->dev = dev;
286 ip->inum = inum;
287 ip->ref = 1;
288 ip->valid = 0;
289 release(&icache.lock);
290
291 return ip;
292}

305// Reads the inode from disk if necessary.
306void ilock(struct inode *ip)
307{
308 struct buf *bp;
309 struct dinode *dip;

312 panic("ilock");
313

315
316 if(ip->valid == 0){
317 bp = bread(ip->dev, IBLOCK(ip->inum, sb));
318 dip = (struct dinode*)bp->data + ip->inum%IPB;
319 ip->type = dip->type;
320 ip->major = dip->major;
321 ip->minor = dip->minor;
322 ip->nlink = dip->nlink;
323 ip->size = dip->size;

325 brelse(bp);

341// Drop a reference to an in-memory inode.
342// If that was the last reference, the inode cache entry can
343// be recycled.
344// If that was the last reference and the inode has no links
345// to it, free the inode (and its content) on disk.
346// All calls to iput() must be inside a transaction in
347// case it has to free the inode.
348void iput(struct inode *ip)
349{
350 acquiresleep(&ip->lock);
351 if(ip->valid && ip->nlink == 0){
352 acquire(&icache.lock);
353 int r = ip->ref;
354 release(&icache.lock);
355 if(r == 1){
356 // inode has no links and no other references: truncate and free.
357 itrunc(ip);
358 ip->type = 0;
359 iupdate(ip);
360 ip->valid = 0;
361 }
362 }
363 releasesleep(&ip->lock);
364
365 acquire(&icache.lock);
366 ip->ref--;
367 release(&icache.lock);
368}

377//PAGEBREAK!
378// Inode content
379//
380// The content (data) associated with each inode is stored
381// in blocks on the disk. The first NDIRECT block numbers
382// are listed in ip->addrs[]. The next NINDIRECT blocks are
383// listed in block ip->addrs[NDIRECT].
384
385// Return the disk block address of the nth block in inode ip.
386// If there is no such block, bmap allocates one.
387static uint bmap(struct inode *ip, uint bn)
388{
389 uint addr, *a;
390 struct buf *bp;
391
392 if(bn < NDIRECT){
393 if((addr = ip->addrs[bn]) == 0)
394 ip->addrs[bn] = addr = balloc(ip->dev);
395 return addr;
396 }
397 bn -= NDIRECT;
398
399 if(bn < NINDIRECT){
400 // Load indirect block, allocating if necessary.
401 if((addr = ip->addrs[NDIRECT]) == 0)
402 ip->addrs[NDIRECT] = addr = balloc(ip->dev);
403 bp = bread(ip->dev, addr);
404 a = (uint*)bp->data;
405 if((addr = a[bn]) == 0){
406 a[bn] = addr = balloc(ip->dev);
407 log_write(bp);
408 }
409 brelse(bp);
410 return addr;
411 }
412
413 panic("bmap: out of range");
414}

416// Truncate inode (discard contents).
417// Only called when the inode has no links
418// to it (no directory entries referring to it)
419// and has no in-memory reference to it (is
420// not an open file or current directory).
421static void itrunc(struct inode *ip)
422{
423 int i, j;
424 struct buf *bp;
425 uint *a;
426
427 for(i = 0; i < NDIRECT; i++){
428 if(ip->addrs[i]){
429 bfree(ip->dev, ip->addrs[i]);
430 ip->addrs[i] = 0;
431 }
432 }
433
434 if(ip->addrs[NDIRECT]){
435 bp = bread(ip->dev, ip->addrs[NDIRECT]);
436 a = (uint*)bp->data;
437 for(j = 0; j < NINDIRECT; j++){
438 if(a[j])
439 bfree(ip->dev, a[j]);
440 }
441 brelse(bp);
442 bfree(ip->dev, ip->addrs[NDIRECT]);
443 ip->addrs[NDIRECT] = 0;
444 }
445
446 ip->size = 0;
447 iupdate(ip);
448}

462// Read data from inode.
463// Caller must hold ip->lock.
464int readi(struct inode *ip, char *dst, uint off, uint n)
465{
466 uint tot, m;
467 struct buf *bp;
468
469 if(ip->type == T_DEV){
470 if(ip->major < 0 || ip->major >= NDEV || !devsw[ip->major].read)
471 return -1;
472 return devsw[ip->major].read(ip, dst, n);
473 }
474
475 if(off > ip->size || off + n < off)
476 return -1;
477 if(off + n > ip->size)
478 n = ip->size - off;
479
480 for(tot=0; tot<n; tot+=m, off+=m, dst+=m){
481 bp = bread(ip->dev, bmap(ip, off/BSIZE));
482 m = min(n - tot, BSIZE - off%BSIZE);
483 memmove(dst, bp->data + off%BSIZE, m);
484 brelse(bp);
485 }
486 return n;
487}

531// Look for a directory entry in a directory.
532// If found, set *poff to byte offset of entry.
533struct inode*
534dirlookup(struct inode *dp, char *name, uint *poff)
535{
536 uint off, inum;
537 struct dirent de;
538
539 if(dp->type != T_DIR)
540 panic("dirlookup not DIR");
541
542 for(off = 0; off < dp->size; off += sizeof(de)){
543 if(readi(dp, (char*)&de, off, sizeof(de)) != sizeof(de))
544 panic("dirlookup read");
545 if(de.inum == 0)
546 continue;
547 if(namecmp(name, de.name) == 0){
548 // entry matches path element
549 if(poff)
550 *poff = off;
551 inum = de.inum;
552 return iget(dp->dev, inum);
553 }
554 }
555
556 return 0;
557}

559// Write a new directory entry (name, inum) into the directory dp.
560int dirlink(struct inode *dp, char *name, uint inum)
561{
562 int off;
563 struct dirent de;
564 struct inode *ip;
565
566 // Check that name is not present.
567 if((ip = dirlookup(dp, name, 0)) != 0){
568 iput(ip);
569 return -1;
570 }
571
572 // Look for an empty dirent.
573 for(off = 0; off < dp->size; off += sizeof(de)){
574 if(readi(dp, (char*)&de, off, sizeof(de)) != sizeof(de))
575 panic("dirlink read");
576 if(de.inum == 0)
577 break;
578 }
579
580 strncpy(de.name, name, DIRSIZ);
581 de.inum = inum;
582 if(writei(dp, (char*)&de, off, sizeof(de)) != sizeof(de))
583 panic("dirlink");
584
585 return 0;
586}

591// Copy the next path element from path into name.
592// Return a pointer to the element following the copied one.
593// The returned path has no leading slashes,
594// so the caller can check *path=='\0' to see if the name is the last one.
595// If no name to remove, return 0.
596//
597// Examples:
598// skipelem("a/bb/c", name) = "bb/c", setting name = "a"
599// skipelem("///a//bb", name) = "bb", setting name = "a"
600// skipelem("a", name) = "", setting name = "a"
601// skipelem("", name) = skipelem("////", name) = 0
602//
603static char* skipelem(char *path, char *name)
604{
605 char *s;
606 int len;
607
608 while(*path == '/')
609 path++;
610 if(*path == 0)
611 return 0;
612 s = path;
613 while(*path != '/' && *path != 0)
614 path++;
615 len = path - s;
616 if(len >= DIRSIZ)
617 memmove(name, s, DIRSIZ);
618 else {
619 memmove(name, s, len);
620 name[len] = 0;
621 }
622 while(*path == '/')
623 path++;
624 return path;
625}

627// Look up and return the inode for a path name.
628// If parent != 0, return the inode for the parent and copy the final
629// path element into name, which must have room for DIRSIZ bytes.
630// Must be called inside a transaction since it calls iput().
631static struct inode* namex(char *path, int nameiparent, char *name)
632{
633 struct inode *ip, *next;
634
635 if(*path == '/')
636 ip = iget(ROOTDEV, ROOTINO);
637 else
638 ip = idup(myproc()->cwd);
639
640 while((path = skipelem(path, name)) != 0){
641 ilock(ip);
642 if(ip->type != T_DIR){
643 iunlockput(ip);
644 return 0;
645 }
646 if(nameiparent && *path == '\0'){
647 // Stop one level early.
648 iunlock(ip);
649 return ip;
650 }
651 if((next = dirlookup(ip, name, 0)) == 0){
652 iunlockput(ip);
653 return 0;
654 }
655 iunlockput(ip);
656 ip = next;
657 }
658 if(nameiparent){
659 iput(ip);
660 return 0;
661 }
662 return ip;
663}
